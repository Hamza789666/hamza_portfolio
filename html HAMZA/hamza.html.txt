<script>
    const ship = document.getElementById('ship');
    const avat = document.getElementById('main-avatar');
    const src = document.getElementById('avatar-src');
    const overlay = document.getElementById('overlay');
    let x = 50;
    let active = false;

    // Aggiunto effetto dissolvenza
    function start() { 
        overlay.style.opacity = '0';
        setTimeout(() => {
            overlay.style.display = 'none';
        }, 1000);
        active = true; 
    }

    document.addEventListener('keydown', (e) => {
        if (!active) return;
        if (e.key === 'ArrowLeft' && x > 5) { x -= 5; ship.style.left = x + '%'; }
        if (e.key === 'ArrowRight' && x < 95) { x += 5; ship.style.left = x + '%'; }
        if (e.key === 'ArrowUp') fire();
    });

    // Generazione Stelle
    for(let i=0; i<150; i++) { // Aumentate per un effetto migliore
        let s = document.createElement('div');
        s.className = 'star';
        s.style.width = s.style.height = Math.random()*3+'px';
        s.style.left = Math.random()*100+'%'; s.style.top = Math.random()*100+'%';
        document.getElementById('universe').appendChild(s);
    }

    function fire() {
        let b = document.createElement('div');
        b.className = 'bullet';
        b.style.left = x + '%'; b.style.bottom = '140px';
        document.getElementById('universe').appendChild(b);
        let p = 140;
        let m = setInterval(() => {
            p += 15; b.style.bottom = p + 'px';
            if(hit(b, 'p-passione')) { clearInterval(m); b.remove(); show('passione'); }
            if(hit(b, 'p-sogno')) { clearInterval(m); b.remove(); show('sogno'); }
            if(hit(b, 'p-competenze')) { clearInterval(m); b.remove(); show('competenze'); }
            if(p > window.innerHeight) { clearInterval(m); b.remove(); }
        }, 10);
    }

    function hit(b, id) {
        let rB = b.getBoundingClientRect();
        let rP = document.getElementById(id).getBoundingClientRect();
        return !(rB.right < rP.left || rB.left > rP.right || rB.bottom < rP.top || rB.top > rP.bottom);
    }

    function show(type) {
        active = false;
        document.querySelectorAll('.content-left').forEach(s => s.style.display = 'none');
        avat.pause();

        if(type === 'passione') {
            document.getElementById('sec-passione').style.display = 'flex';
            src.src = "Adam - Staff.mp4";
            document.getElementById('status-msg').innerText = "Area Passioni Caricata.";
        } else if(type === 'sogno') {
            document.getElementById('sec-sogno').style.display = 'flex';
            src.src = "video_sogno.mp4";
            document.getElementById('status-msg').innerText = "Sogni e Obiettivi.";
        } else {
            document.getElementById('sec-competenze').style.display = 'flex';
            src.src = "nuovo_avatar 2.mp4";
            document.getElementById('status-msg').innerText = "Competenze Tecniche AI.";
        }

        avat.load();
        document.getElementById('universe').style.opacity = '0';
        document.getElementById('gallery').style.display = 'flex';
        
        avat.oncanplay = () => {
            avat.play().catch(() => { avat.muted = true; avat.play(); });
            if(type === 'passione') {
                const v = document.getElementById('vid-pass');
                v.play().catch(() => {});
            }
        };
    }

    function closeAll() {
        document.getElementById('gallery').style.display = 'none';
        document.getElementById('universe').style.opacity = '1';
        avat.pause();
        active = true;
    }
</script>
